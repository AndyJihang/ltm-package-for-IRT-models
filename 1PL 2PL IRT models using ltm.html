rel.ltm <- function(x) {
    eap <- ltm::factor.scores(x, method="EAP")$score.dat
    e <- sum(eap$Obs*eap$se.z1^2)/sum(eap$Obs)
    s <- sum(eap$Obs*eap$z1^2)/sum(eap$Obs)
    1-(e/(s+e))
}
